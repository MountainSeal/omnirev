\begin{prooftree}
  \AxiomC{}
  \RightLabel{identity}
  \UnaryInfC{$x:\tau \vdash x:\tau$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash \tau:type$}
  \AxiomC{$\Delta \vdash x:\tau$}
  \RightLabel{annotation}
  \BinaryInfC{$\Gamma,\Delta \vdash (x::\tau):\tau$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma,y:\tau,x:\sigma,\Delta \vdash z:\upsilon$}
  \RightLabel{exchange}
  \UnaryInfC{$\Gamma,x:\sigma,y:\tau,\Delta \vdash z:\upsilon$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash t:\tau$}
  \RightLabel{unit${}_\text{L}$}
  \UnaryInfC{$\Gamma,x:unit \vdash t:\tau$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{}
  \RightLabel{unit${}_\text{R}$}
  \UnaryInfC{$\quad \vdash \langle\rangle:unit$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma,x:\sigma \vdash t:\tau$}
  \RightLabel{abstraction}
  \UnaryInfC{$\Gamma \vdash lambda\;x::\sigma.t:forall\;x::\sigma.\tau$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash f:forall\;x::\sigma.\tau$}
  \AxiomC{$\Delta \vdash t:\sigma$}
  \RightLabel{application}
  \BinaryInfC{$\Gamma,\Delta \vdash (f\;t):\tau$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash t:\tau$}
  \AxiomC{$\cdots$}
  \AxiomC{$\Delta \vdash u:\upsilon$}
  \RightLabel{pair}
  \TrinaryInfC{$\Gamma,\cdots,\Delta \vdash \{t,\cdots,u\}:\{\tau*\cdots*\upsilon\}$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma,x_1:\tau_1,\cdots,x_n:\tau_n \vdash u:\upsilon$}
  \AxiomC{$\Delta \vdash t:\{\tau_1*\cdots*\tau_n\}$}
  \RightLabel{let}
  \BinaryInfC{$\Gamma,\Delta \vdash let\;\{x_1,\cdots,x_n\}=t\;in\;u:\upsilon$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash x:\tau_i$}
  \RightLabel{injection${}_\text{i}$}
  \UnaryInfC{$\Gamma \vdash \langle{c_i\;x}\rangle:c_1\;of\;\tau_1+\dots+c_i\;of\;\tau_i+\dots+c_n\;of\;\tau_n$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma,x_1:\tau_1 \vdash u_1:\upsilon$}
  \AxiomC{$\cdots$}
  \AxiomC{$\Gamma,x_n:\tau_n \vdash u_n:\upsilon$}
  \AxiomC{$\Delta \vdash t:c_1\;of\;\tau_1+\dots+c_i\;of\;\tau_i+\dots+c_n\;of\;\tau_n$}
  \RightLabel{case}
  \QuaternaryInfC{$\Gamma,\Delta \vdash case\;t\;of\;\langle{c_1\;x_1}\rangle\,\Rightarrow\,u_1\;|\cdots\;|\langle{c_n\;x_n}\rangle\,\Rightarrow\,u_n:\upsilon$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{}
  \RightLabel{type}
  \UnaryInfC{$\quad \vdash type:type$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{}
  \RightLabel{unit}
  \UnaryInfC{$\quad \vdash unit:type$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash (x::\sigma):\sigma$}
  \AxiomC{$\Delta \vdash \tau:type$}
  \RightLabel{forall}
  \BinaryInfC{$\Gamma,\Delta \vdash forall\;x::\sigma.\tau:type$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash \sigma:type$}
  \AxiomC{$\cdots$}
  \AxiomC{$\Delta \vdash \tau:type$}
  \RightLabel{product}
  \TrinaryInfC{$\Gamma,\cdots,\Delta \vdash \{\sigma*\cdots*\tau\}:type$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash \tau_1:type$}
  \AxiomC{$\cdots$}
  \AxiomC{$\Delta \vdash \tau_n:type$}
  \RightLabel{sum}
  \TrinaryInfC{$\Gamma,\cdots,\Delta \vdash c_1\;of\;\tau_1+\dots+c_i\;of\;\tau_i+\dots+c_n\;of\;\tau_n:type$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma,x:\tau \vdash t:\tau$}
  \RightLabel{fixpoint(これいら無くない?)}
  \UnaryInfC{$\Gamma \vdash fix\;x::\tau.t:\tau$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash x:\tau$}
  \RightLabel{duplication}
  \UnaryInfC{$\Gamma \vdash |x|:\{\tau*\tau\}$}
\end{prooftree}

\begin{prooftree}
  \AxiomC{$\Gamma \vdash x:\{\tau*\tau\}$}
  \RightLabel{equality}
  \UnaryInfC{$\Gamma \vdash |x|:\tau$}
\end{prooftree}
