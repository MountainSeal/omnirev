\begin{itembox}[c]{Type Formation rules}
  \begin{center}
    \begin{bprooftree}
    \AxiomC{}
      %\RightLabel{$I$}
      \UnaryInfC{$\sequent{\Gamma}{I}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{T_1}$}
      \AxiomC{$\sequent{\Gamma}{T_2}$}
      %\RightLabel{$\oplus$}
      \BinaryInfC{$\sequent{\Gamma}{T_1\oplus{}T_2}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{T_1}$}
      \AxiomC{$\sequent{\Gamma}{T_2}$}
      %\RightLabel{$\otimes$}
      \BinaryInfC{$\sequent{\Gamma}{T_1\otimes{}T_2}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{T_1}$}
      \AxiomC{$\sequent{\Gamma}{T_2}$}
      %\RightLabel{$\invamp$}
      \BinaryInfC{$\sequent{\Gamma}{T_1\invamp{}T_2}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{T_1}$}
      \AxiomC{$\sequent{\Gamma}{T_2}$}
      %\RightLabel{$\multimap$}
      \BinaryInfC{$\sequent{\Gamma}{T_1\multimap{}T_2}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{}
      %\RightLabel{Variable}
      \UnaryInfC{$\sequent{\Gamma,X}{X}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma,X}{T}$}
      %\RightLabel{$\mu$}
      \UnaryInfC{$\sequent{\Gamma}{\mu{X}.T}$}
    \end{bprooftree}
  \end{center}
\end{itembox}

\begin{itembox}[c]{Typing rules for Values}
  \begin{prooftree}
    \AxiomC{}
    \RightLabel{Variable}
    \UnaryInfC{$\sequent{x:T}{x:T}$}
  \end{prooftree}
  \begin{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{\Delta}$}
      \LeftLabel{$I_L$}
      \UnaryInfC{$\sequent{\Gamma,():I}{\Delta}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{\Delta}$}
      \RightLabel{$I_R$}
      \UnaryInfC{$\sequent{\Gamma}{():I,\Delta}$}
    \end{bprooftree}
  \end{center}
  \begin{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma,v_1:T_1}{\Delta}$}
      \LeftLabel{$\oplus_{L_l}$}
      \UnaryInfC{$\sequent{\Gamma,inl\;v_1:T_1\oplus{}T_2}{\Delta}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{v_1:T_1,\Delta}$}
      \RightLabel{$\oplus_{R_l}$}
      \UnaryInfC{$\sequent{\Gamma}{inl\;v_1:T_1\oplus{}T_2,\Delta}$}
    \end{bprooftree}
  \end{center}
  \begin{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma,v_2:T_2}{\Delta}$}
      \LeftLabel{$\oplus_{L_r}$}
      \UnaryInfC{$\sequent{\Gamma,inr\;v_2:T_1\oplus{}T_2}{\Delta}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{v_2:T_2,\Delta}$}
      \RightLabel{$\oplus_{R_r}$}
      \UnaryInfC{$\sequent{\Gamma}{inr\;v_2:T_1\oplus{}T_2,\Delta}$}
    \end{bprooftree}
  \end{center}
  \begin{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma,v_1:T_1,v_2:T_2}{\Delta}$}
      \LeftLabel{$\otimes_L$}
      \UnaryInfC{$\sequent{\Gamma,v_1\times{}v_2:T_1\otimes{}T_2}{\Delta}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma_1}{v_1:T_1,\Delta_1}$}
      \AxiomC{$\sequent{\Gamma_2}{v_2:T_2,\Delta_2}$}
      \RightLabel{$\otimes_R$}
      \BinaryInfC{$\sequent{\Gamma_1;\Gamma_2}{v_1\times{}v_2:T_1\otimes{}T_2,\Delta_1;\Delta_2}$}
    \end{bprooftree}
  \end{center}
  \begin{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma_1,v_1:T_1}{\Delta_1}$}
      \AxiomC{$\sequent{\Gamma_2,v_2:T_2}{\Delta_2}$}
      \LeftLabel{$\invamp_L$}
      \BinaryInfC{$\sequent{\Gamma_1;\Gamma_2,v_1\parallel{}v_2:T_1\invamp{}T_2}{\Delta_1;\Delta_2}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{v_1:T_1,v_2:T_2,\Delta}$}
      \RightLabel{$\invamp_R$}
      \UnaryInfC{$\sequent{\Gamma}{v_1\parallel{}v_2:T_1\invamp{}T_2,\Delta}$}
    \end{bprooftree}
  \end{center}
  \begin{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma_1}{v_1:T_1,\Delta_1}$}
      \AxiomC{$\sequent{\Gamma_2,v_2:T_2}{\Delta_2}$}
      \LeftLabel{$\multimap_L$}
      \BinaryInfC{$\sequent{\Gamma_1;\Gamma_2,v_1\mapsto{}v_2:T_1\multimap{}T_2}{\Delta_1;\Delta_2}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma,v_1:T_1}{v_2:T_2,\Delta}$}
      \RightLabel{$\multimap_R$}
      \UnaryInfC{$\sequent{\Gamma}{v_1\mapsto{}v_2:T_1\multimap{}T_2,\Delta}$}
    \end{bprooftree}
  \end{center}
  \begin{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma,v:T[\mu{X}.T/X]}{\Delta}$}
      \LeftLabel{$\mu_L$}
      \UnaryInfC{$\sequent{\Gamma,fold\;v:\mu{X}.T}{\Delta}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{\Gamma}{v:T[\mu{X}.T/X],\Delta}$}
      \RightLabel{$\mu_R$}
      \UnaryInfC{$\sequent{\Gamma}{fold\;v:\mu{X}.T,\Delta}$}
    \end{bprooftree}
  \end{center}
\end{itembox}

\begin{itembox}[c]{Typing rules for Terms}
  \begin{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{}{v_2:T_1\multimap{}T_2}$}
      \AxiomC{$\sequent{}{v_1:T_1}$}
      \LeftLabel{Application}
      \BinaryInfC{$\sequent{}{v_2\;v_1:T_2}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{}{v_1:T_1\multimap{}T_2}$}
      \AxiomC{$\sequent{}{v_2:T_2\multimap{}T_3}$}
      \RightLabel{Composition}
      \BinaryInfC{$\sequent{}{v_1\fcmp{}v_2:T_1\multimap{}T_3}$}
    \end{bprooftree}
  \begin{center}
  \end{center}
    \begin{bprooftree}
      \AxiomC{$\sequent{}{v:T_2\multimap{}T_1}$}
      \LeftLabel{Transpose}
      \UnaryInfC{$\sequent{}{v^\dagger{}:T_1\multimap{}T_2}$}
    \end{bprooftree}
    \begin{bprooftree}
      \AxiomC{$\sequent{}{v:T_1\invamp{}T_2}$}
      \RightLabel{Measurement}
      \UnaryInfC{$\sequent{}{measure\;v:T_1\oplus{}T_2}$}
    \end{bprooftree}
  \end{center}
\end{itembox}

\begin{itembox}[c]{Substitution}
  \begin{align*}
    Y[S/X] &=
    \begin{cases}
      S & \text{if}Y = X \\
      Y & \text{otherwise}
    \end{cases} \\
    I[S/X] &= I \\
    T_1\oplus{}T_2[S/X] &= T_1[S/X]\oplus{}T_2[S/X] \\
    T_1\otimes{}T_2[S/X] &= T_1[S/X]\otimes{}T_2[S/X] \\
    T_1\invamp{}T_2[S/X] &= T_1[S/X]\invamp{}T_2[S/X] \\
    T_1\multimap{}T_2[S/X] &= T_1[S/X]\multimap{}T_2[S/X] \\
    \mu{Y}.T[S/X] &= \mu{Y}.(T[S/X])
    %再帰型の入れ子は型付けで許さないのでここの定義は修正した方がいい鴨
  \end{align*}
\end{itembox}
